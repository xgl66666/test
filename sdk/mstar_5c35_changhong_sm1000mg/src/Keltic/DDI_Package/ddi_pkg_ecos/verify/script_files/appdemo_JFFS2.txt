# need to init flash first if not init before.

# mount flash to /fs at 5MB addr with 1MB size
# if mount fail, make sure the flash size or need to erase the used flash. 
# ex. MDrv_SERFLASH_AddressErase(0x500000, 0x600000 - 1, true);
jffs2_mount /dev/flash/0 /fs 0x500000 0x100000

# get free sectors, need to keep more than GC overhead sectors when writing data to flash via jffs2. 
jffs2_getfreesectors /fs

#umount device
usb_umount /fs
