#Toggle SCART bypass control from VCR SCART to TV SCART
#set scart bypass on(need two scart device in HW Board. One for output, the other for input)

#set SCART bypass off
Scart_Bypass 0

#set SCART Output Mode, 0 -> CVBS+SCART-RGB on, 1 -> all disable, 2 -> CVBS only
Scart_SetOutputMode 2

#set SCART out Aspect Ratio, 0 -> 1to1, 1 -> 16to9, 2 -> 4to3
Scart_SetAspectRatio 1

#Set SCART interrupt enable or not, 0 -> disable, 1 -> enable
Scart_SetInterruptEnable 1

#Get SCART interrupt flag and clear it
Scart_GetInterruptOut

#Set SCART output audio level adjustment (0 to -62db, -2db step)
Scart_SetAudioLevelAdjustment 15
