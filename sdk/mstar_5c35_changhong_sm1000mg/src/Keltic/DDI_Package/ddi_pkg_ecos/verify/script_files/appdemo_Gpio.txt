#initial gpio
gpio_init 

#  set gpio10 to output and no invert and output = off
gpio_set 10 0 0 0   

# get the level of gpio 10
gpio_getlevel 10  

#  set gpio10 to output and no invert and output = on
gpio_set 10 0 0 1   

# get the level of gpio 10
gpio_getlevel 10 


#  set gpio10 to input
gpio_set 10 1 0 0   

# get the level of gpio 10
gpio_getlevel 10
