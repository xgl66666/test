#timeshift 
#First, run watchtv script

#not freeze screen

PVR_TimeshiftRecordStart 801 802 801 0 1 timeshift.ts

#Third, timeshift playback command
PVR_TimeshiftPlaybackStart

PVR_GetTimeshiftRecStartTime

#jump backward and then normal play

PVR_TimeshiftPlaybackStop
# stop recording and playback
